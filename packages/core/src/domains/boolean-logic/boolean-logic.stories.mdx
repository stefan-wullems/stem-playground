import { Meta, Story, Preview } from '@storybook/addon-docs/blocks'
import { useState } from 'react'
import { DeductiveArgumentTruthTable } from './DeductiveArgumentTruthTable' 

<Meta title="BooleanLogic" />

# Deductive Argument Truth Table

<Preview>
  <Story name='Deductive argument truth table'>
    {() => {
      const [draftPropositions, setDraftPropositions] = useState("(J|B)&(!B|!S);!J");
      const [draftConclusions, setDraftConclusions] = useState("!S;B");
      const [propositions, setPropositions] = useState([]);
      const [conclusions, setConclusions] = useState([]);
      function commitInputs() {
        setPropositions(draftPropositions.split(';').map(s => s.trim()))
        setConclusions(draftConclusions.split(';').map(s => s.trim()))
      }
      React.useEffect(() => {commitInputs()}, [])
      return (
        <div>
          <label>
            Propositions csv
            <input
              value={draftPropositions}
              onChange={(e) => setDraftPropositions(e.target.value)}
            />
          </label>
          <label>
            Conclusions csv
            <input
              value={draftConclusions}
              onChange={(e) => setDraftConclusions(e.target.value)}
            />
          </label>
          <button onClick={() => commitInputs()}>
            Generate
          </button>
          <DeductiveArgumentTruthTable propositions={propositions} conclusions={conclusions} />
        </div>
      )
    }}
  </Story>
</Preview>
